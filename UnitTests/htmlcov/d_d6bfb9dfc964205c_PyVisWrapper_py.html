260 def create_nodes(self, g, list_of_objects): 

261

263 list_of_objects = remove_duplicates_in_iterable_based_on_asset_id(list_of_objects)

265 nodes = [] 

266 index = 0

268 naam = otl_object.__class__.__name__ + '_' + otl_object.assetId.identificator[0:36] 

269 if hasattr(otl_object, 'naam'):

271 

272 selected_color = self.random_color_if_not_in_dict(otl_object.typeURI) 

273 

274 tooltip = self.get_tooltip(otl_object)

277 shape = 'square' 

278 

279 g.add_node(otl_object.assetId.identificator,
